\define@key{edukeys}{loc}{\def\edu@loc{#1}}
\define@key{edukeys}{startdate}{\def\edu@startdate{#1}}
\define@key{edukeys}{enddate}{\def\edu@enddate{#1}}
\define@key{edukeys}{degree}{\def\edu@degree{#1}}
\define@key{edukeys}{major}{\def\edu@major{#1}}
\define@key{edukeys}{minor}{\def\edu@minor{#1}}
\define@key{edukeys}{gpa}{\def\edu@gpa{#1}}
\define@key{edukeys}{courses}{\def\edu@courses{#1}}

\newcommand{\edu}[2]{%
    \begingroup%
    \setkeys{edukeys}{loc=, startdate=, enddate=, degree=, major=, minor=, gpa=, courses=, #2}%
    \def\mmg{% major, minor, gpa
        \ifx\edu@major\empty{%
            \ifx\edu@gpa\empty\else{GPA: \edu@gpa}\fi}%
        \else{%
            Major: \edu@major%
            \ifx\edu@minor\empty\else{; Minor: \edu@minor}\fi%
            \ifx\edu@gpa\empty\else{; GPA: \edu@gpa}\fi}%
        \fi%
    }
    \textbf{#1}\hfill\edu@loc\par
    \ifx\edu@degree\empty{\mmg}\else{\edu@degree}\fi%
    \hfill
    \ifx\edu@startdate\empty\edu@enddate\else{\edu@startdate\,--\,\edu@enddate}\fi\par
    \ifx\edu@degree\empty\else{\mmg\par}\fi%
    \ifx\edu@courses\empty\else{Courses: \edu@courses\par}\fi
    \endgroup\par\addvspace{\postentryskip}
}

\define@key{activitykeys}{role}{\def\activity@role{#1}}
\define@key{activitykeys}{org}{\def\activity@org{#1}}
\define@key{activitykeys}{loc}{\def\activity@loc{#1}}
\define@key{activitykeys}{startdate}{\def\activity@startdate{#1}}
\define@key{activitykeys}{enddate}{\def\activity@enddate{#1}}
\define@key{activitykeys}{hoursperweek}{\def\activity@hoursperweek{#1}}
\define@key{activitykeys}{weeksperyear}{\def\activity@weeksperyear{#1}}

\newenvironment{activity}[1]{%
    \begingroup%
    \setkeys{activitykeys}{org=, loc=, startdate=, enddate=, hoursperweek=, weeksperyear=, #1}%
    \textbf{\emph{\activity@role}\ifx\activity@org\empty\else{, \activity@org}\fi}%
    <! if settings.show_activity_locations !>\hfill\activity@loc<! endif !>\par
    \expandafter\ifstrempty\expandafter{\activity@startdate}{\activity@enddate}{\activity@startdate\,--\,\activity@enddate}%
    <! if settings.show_time_commitments !>\hfill
    \expandafter\ifstrempty\expandafter{\activity@hoursperweek}{}{%
        \activity@hoursperweek\ \expandafter\ifstrequal\expandafter{\activity@hoursperweek}{1}{hour}{hours}/week}%
    \expandafter\ifstrempty\expandafter{\activity@hoursperweek}{}{%
        \expandafter\ifstrempty\expandafter{\activity@weeksperyear}{}{,}}
    \expandafter\ifstrempty\expandafter{\activity@weeksperyear}{}{%
        \activity@weeksperyear\ \expandafter\ifstrequal\expandafter{\activity@weeksperyear}{1}{week}{weeks}/year}%
    <! endif !>\par
    \endgroup
    \begin{itemize}[nosep]
}{%
    \end{itemize}\par\addvspace{\postentryskip}
}
\makeatother
